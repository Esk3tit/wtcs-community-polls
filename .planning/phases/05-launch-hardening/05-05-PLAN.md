    ```typescript
    import { createClient } from '@supabase/supabase-js'
    import type { Page } from '@playwright/test'

    const SUPABASE_URL = process.env.VITE_SUPABASE_URL ?? 'http://localhost:54321'
    const SERVICE_ROLE = process.env.SUPABASE_SERVICE_ROLE_KEY
    if (!SERVICE_ROLE) {
      throw new Error('SUPABASE_SERVICE_ROLE_KEY env var required for Playwright auth helper')
    }

    // Pitfall 7: storage key is `sb-<project-ref>-auth-token`. For `supabase start`,
    // ref is the hostname prefix of the API URL. Verify against the running app's
    // DevTools → Application → Local Storage if tests fail to authenticate.
    const PROJECT_REF = new URL(SUPABASE_URL).hostname.split('.')[0] || 'localhost'
    const STORAGE_KEY = `sb-${PROJECT_REF}-auth-token`

    export async function loginAs(page: Page, fixtureUserId: string): Promise<void> {
      const admin = createClient(SUPABASE_URL, SERVICE_ROLE, {
        auth: { autoRefreshToken: false, persistSession: false },
      })

      // Mint session for the fixture user via admin.auth.admin.generateLink or
      // admin.auth.admin.signInWithUserId if available on the Supabase version.
      // For supabase-js 2.101.1, use the access-token approach:
      const { data, error } = await admin.auth.admin.getUserById(fixtureUserId)
      if (error || !data?.user) throw error ?? new Error(`No fixture user ${fixtureUserId}`)

      // Generate a signed JWT for the user. Simplest: use signInWithPassword with a
      // fixture password set in the seed, OR use generateLink to obtain tokens.
      const { data: linkData, error: linkErr } = await admin.auth.admin.generateLink({
        type: 'magiclink',
        email: data.user.email!,
      })
      if (linkErr) throw linkErr

      const session = {
        access_token: linkData.properties.action_link.split('access_token=')[1]?.split('&')[0],
        refresh_token: linkData.properties.action_link.split('refresh_token=')[1]?.split('&')[0],
        user: data.user,
        // ... additional session fields as needed
      }

      await page.addInitScript(
        ([key, payload]) => {
          window.localStorage.setItem(key, JSON.stringify(payload))
        },
        [STORAGE_KEY, session] as [string, typeof session],
      )
    }
    ```

    Note: if `generateLink` token extraction proves brittle, fall back to
    `admin.auth.admin.createUser(...)` with a known password + `signInWithPassword`
    to get a real session. Document the chosen path in a comment.
