  localStorage consent check. Users who opt out never load the Replay bundle.

  Task 2 via `npm run build` size check.

    **src/lib/sentry.ts** (NEW — M1 resolution) — consent-gated lazy Replay
    loader. No Replay code lives in main.tsx's Sentry.init anymore.

    ```typescript
    import * as Sentry from '@sentry/react'

    let replayLoaded = false

    /**
     * Lazily attaches Sentry Replay to the running client when the user has
     * not opted out of analytics. Called from ConsentChip's mount effect
     * AFTER the localStorage consent check (M1 — prevents first-paint capture
     * for opt-out users; also code-splits the replay bundle per M3).
     */
    export async function loadSentryReplayIfConsented(): Promise<void> {
      if (replayLoaded || typeof window === 'undefined') return
      // Respect the same localStorage flag ConsentChip uses.
      const dismissed = window.localStorage.getItem('posthog_consent_chip_dismissed') === 'true'
      // Phase 5 semantics (UI-SPEC Contract 3): dismissed-via-Opt-out implies no analytics.
      // Dismissed-via-X (accept) is fine — chip just hides; analytics continue.
      // We can't distinguish the two from localStorage alone, so we use a second flag set by ConsentChip's Opt-out handler.
      const optedOut = window.localStorage.getItem('analytics_opted_out') === 'true'
      if (optedOut) return
      const client = Sentry.getClient()
      if (!client) return
      const { replayIntegration } = await import('@sentry/react')
      client.addIntegration(
        replayIntegration({
          maskAllText: true,
          blockAllMedia: true,
        }),
      )
      replayLoaded = true
    }
    ```
