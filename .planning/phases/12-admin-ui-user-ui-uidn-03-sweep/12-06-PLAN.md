    ```ts
    /**
     * castVoteOnFreshPoll — TEST-13 helper for the admin-hide/voter-sees-Alert spec.
     *
     * Inserts one `votes` row keyed on the given user via the admin (service-role)
     * client because `votes` has no INSERT RLS policy — production writes go
     * through the `submit-vote` Edge Function. Tests bypass the EF for setup
     * speed; the EF path is exercised by separate specs.
     *
     * Returns the choice id that was voted on so the test can verify the
     * voter-side "Your response: {label}" line in the hidden-state branch.
     */
    export async function castVoteOnFreshPoll(opts: {
      pollId: string
      userId: string
    }): Promise<{ choiceId: string; label: string }> {
      const admin = getAdminClient()
      // The freshPoll fixture inserts two choices labelled 'Yes' (sort_order 1)
      // and 'No' (sort_order 2). Pick the first (Yes) for determinism.
      const { data: choices, error: choicesErr } = await admin
        .from('choices')
        .select('id, label')
        .eq('poll_id', opts.pollId)
        .order('sort_order', { ascending: true })
        .limit(1)
      if (choicesErr || !choices || choices.length === 0) {
        throw choicesErr ?? new Error('castVoteOnFreshPoll: no choices found for poll')
      }
      const target = choices[0]
      const { error: voteErr } = await admin.from('votes').insert({
        poll_id: opts.pollId,
        choice_id: target.id,
        user_id: opts.userId,
      })
      if (voteErr) throw voteErr
      return { choiceId: target.id, label: target.label }
    }
    ```

    The helper is purely a setup utility — the spec calls it once before navigating the voter to `/topics`. Cleanup: the freshPoll teardown cascades to `votes` via the existing FK ON DELETE CASCADE, so no additional cleanup logic is needed for the vote row.

    import { loginAs } from '../helpers/auth'

    test('[@smoke] admin hide/show roundtrip — voter sees Alert on hide, ResultBars on show', async ({
      page,
      browser,
      freshPoll,
    }) => {
      // STEP 1: Pre-cast a vote for memberUser on freshPoll (avoids re-implementing
      // the full submit-vote click path; the vote-cast flow is covered by browse-respond.spec.ts).

        pollId: freshPoll.id,
        userId: fixtureUsers.memberUser.id,
      })

      // STEP 2: Voter session in the default context. Log in as memberUser and navigate
      // to /topics. Verify the suggestion card is visible and currently shows ResultBars
      // (results_hidden = false by default at create).

      await page.goto('/topics')
      const voterCard = page
        .locator('article, [data-testid^="suggestion-card"]')
        .filter({ hasText: freshPoll.title })
        .first()
      await expect(voterCard).toBeVisible({ timeout: 10_000 })
      // The voter has voted; the hidden-state Alert is NOT yet visible.
      await expect(
        page.getByTestId(`results-hidden-alert-${freshPoll.id}`),
      ).not.toBeVisible()

      // STEP 3: Admin session in a SECOND browser context. Log in as adminUser, go to
      // /admin (defaults to the suggestions tab), find the row, click the Switch.
      const adminContext = await browser.newContext()

        await loginAs(adminPage, fixtureUsers.adminUser.id)

        // STEP 4: Voter side polls every 8s. Within ~12s the Alert appears.
        await expect(
          page.getByTestId(`results-hidden-alert-${freshPoll.id}`),

        await expect(page.locator(`[data-testid="results-hidden-alert-${freshPoll.id}"]`)).toContainText('Your response')
