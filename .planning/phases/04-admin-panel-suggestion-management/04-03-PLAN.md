    ```typescript
    import { useState, useCallback } from 'react'
    import { supabase } from '@/lib/supabase'
    import { toast } from 'sonner'

    function unwrapError(error: unknown, fallback: string): string {
      try {
        const ctx = (error as { context?: { json?: () => Promise<{ error?: string }> } }).context
        if (ctx?.json) {
          // best-effort sync unwrap not possible; caller handles async
        }
      } catch { /* fall through */ }
      return fallback
    }

    export function useCategoryMutations() {
      const [submitting, setSubmitting] = useState(false)

      const create = useCallback(async (name: string) => {
        setSubmitting(true)
        try {
          const { data, error } = await supabase.functions.invoke<{ id: string; name: string }>(
            'create-category', { body: { name } }
          )
          if (error) {
            let msg = 'Could not create category. Try again.'
            try {
              const body = await (error as any).context?.json?.()
              if (body?.error) msg = body.error
            } catch { /* fall through */ }
            toast.error(msg)
            return { ok: false as const }
          }
          toast.success('Category created')
          return { ok: true as const, category: data }
        } finally { setSubmitting(false) }
      }, [])

      const rename = useCallback(async (category_id: string, name: string) => {
        setSubmitting(true)
        try {
          const { error } = await supabase.functions.invoke('rename-category', { body: { category_id, name } })
          if (error) {
            let msg = 'Could not rename category. Try again.'
            try {
              const body = await (error as any).context?.json?.()
              if (body?.error) msg = body.error
            } catch { /* fall through */ }
            toast.error(msg)
            return { ok: false as const }
          }
          toast.success('Category renamed')
          return { ok: true as const }
        } finally { setSubmitting(false) }
      }, [])

      const remove = useCallback(async (category_id: string, affectedCount: number = 0) => {
        setSubmitting(true)
        try {
          const { error } = await supabase.functions.invoke('delete-category', { body: { category_id } })
          if (error) {
            toast.error('Could not delete category. Try again.')
            return { ok: false as const }
          }
          toast.success(`Category deleted. ${affectedCount} suggestions are now uncategorized.`)
          return { ok: true as const }
        } finally { setSubmitting(false) }
      }, [])

      return { create, rename, remove, submitting }
    }
    ```

    Apply the same pattern to `usePromoteAdmin.ts`, `useDemoteAdmin.ts`, `useSearchAdminTargets.ts`. For useDemoteAdmin, special-case the 400 "Cannot demote yourself" by checking the unwrapped error string. For useSearchAdminTargets, use existing `useDebounce`:

    ```typescript
    import { useState, useEffect } from 'react'
    import { supabase } from '@/lib/supabase'
    import { useDebounce } from './useDebounce'

    type Target = { id: string; discord_id: string; discord_username: string; avatar_url: string | null }

    export function useSearchAdminTargets() {
      const [query, setQuery] = useState('')
      const [results, setResults] = useState<Target[]>([])
      const [searching, setSearching] = useState(false)
      const debounced = useDebounce(query, 300)

      useEffect(() => {
        if (debounced.length < 2) { setResults([]); return }
        let cancelled = false
        setSearching(true)
        supabase.functions.invoke<{ results: Target[] }>('search-admin-targets', { body: { query: debounced } })
          .then(({ data, error }) => {
            if (cancelled) return
            if (error || !data) { setResults([]); return }
            setResults(data.results ?? [])
          })
          .finally(() => { if (!cancelled) setSearching(false) })
        return () => { cancelled = true }
      }, [debounced])

      return { query, setQuery, results, searching }
    }
    ```

    ```tsx
          </div>
          <div className="divide-y">
            {admins.map((a) => {
              const isSelf = user?.id === a.id
              return (
    ```
